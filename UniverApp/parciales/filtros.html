Filtros

Podemos usar los filtros nativos del core de AngularJS (ver Ejercicio1) o crear nuestros propios filtros, como estos que he ido creando: 



        // Filtro para poner la primera letra en Mayusculas de un texto
        app.filter('capitalizar', function(){ 

            return function(texto){
        
                return texto.charAt(0).toUpperCase() + texto.slice(1);
        
            }
        
        });

Frase orginal: "hoy es viernes"

Aplicando filtro -> {{viernes | capitalizar}}


        //filtro para poner en mayus las partes que queramos
        app.filter('capitalizarCacho', function(){

            return function(texto, inicio, fin){
          
              return texto.slice(0, inicio) + texto.slice(inicio,fin).toUpperCase()+ texto.slice(fin);
          
            }
          });

Frase orginal: "hoy es viernes"

Aplicando filtro -> {{viernes | capitalizarCacho: 7: 10}}



        // Filtro para poner una cantidad en euros con formato de esp
        app.filter('euros', function(){ 

            return function(numero){
          
                return numero.replace(".", ",") +" €";
          
            }
          
          });
